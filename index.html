Czyste funkcje


const x = 1
const y = 2

const impureFunction = () => x + y
const pureFunction = (x, y) => x + y

Czyste funkcje


const o = { a: 1 }

const impureFunction = () => o.a++
const impureFunctionEither = o => o.a++
const pureFunction = o =>
	Object.assign({}, o, { a: o.a + 1 })

Czyste funkcje


const o = { a: 1 }

function impureFunction() {
	return o.a++
}
function impureFunctionEither(o) {
	o.a++
	return o
}
function pureFunction(o) {
	return {...o, a: o.a + 1 }
}

Mutacje


const obj = { a: 1 }
const arr = [ 3, 1, 2 ]

// mutation
obj.a = 2
arr.sort()

// immutable
const newObj = { ...obj, a: 2 }
const newArr = [...arr].sort()

Mutacje



const a = [ 3, 1, 2 ]
let sum = 0

// mutation
a.forEach((el, i, arr) => arr[i] = el * 2)
a.forEach(el => sum += el)

// immutable
const newA = a.map(el => el * 2)
const newSum = a.reduce((a, b) => a + b)

Currying


const add = a => b => a + b
add(1)(2) === 3
